2、赋予root权限
 方法一：修改 /etc/sudoers 文件，找到下面一行，把前面的注释（#）去掉
 ## Allows people in group wheel to run all commands
   %wheel    ALL=(ALL)    ALL
  然后修改用户，使其属于root组（wheel），命令如下：
  # usermod -g root tommy
 修改完毕，现在可以用tommy帐号登录，然后用命令 su - ，即可获得root权限进行操作。
 方法二：修改 /etc/sudoers 文件，找到下面一行，在root下面添加一行，如下所示：
 ## Allow root to run any commands anywhere
  root    ALL=(ALL)     ALL
  tommy   ALL=(ALL)     ALL
 修改完毕，现在可以用tommy帐号登录，然后用命令 su - ，即可获得r      oot权限进行操作。

// j 向下
// l 向右

// ^ 0 当前行的行首
// w 跳到下一个单词
// b 跳到上一个单词
// % 跳到对应的另一半括号
// { 上一个空白行
// } 下一个空白行

// nx 删除n个字母 n缺省值是1
// ndw 删除n个单词,n缺省值是1
// u 撤销
// db 删除光标所在位置的前一个开始，到单词首
// D 光标所在位置，到当前行的行尾
// dG 删除光标所在行到文件结尾
// d} 删除当前行到下一个空白行

// yy 复制当前行
// p 粘贴
// :r file 将file中的内容全部倒入到光标所在位置
// :vsp file 将屏幕垂直分隔 ctrl + ww

void diamond(int r) {
int x, y;

for (y=0; y<=2*r; y++) {
    for (x=0; x<=2*r; x++) {
        if ( y==x+r || y==x-r || 
                y==-x+r || y==-x+3*r )
            printf("*");
        else
            printf(" ");
    }
    printf("\n");
}

}

int main( void ) {
int r;

printf("r= ");
scanf("%d", &r);

diamond(r);

}
