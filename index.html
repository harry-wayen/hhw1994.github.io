#include

#include

#include
using namespace std;

#define Max 32767

typedef struct
{
char value;
int lchild,rchild,parent;
int weight;
}HTNode,HuffTree;
typedef char *HuffCode;

void CreatHuffTree(HuffTree &HT,int w,char a,int n)
{
HT=(HuffTree )malloc(sizeof(HTNode)(2n-1));
int i=0,j=0;
int s1 =0, s2=0;
int min1 = Max,min2 = Max;
for(i=0;i<n;++i)//初始化
{
HT[i].value=a[i];
HT[i].weight=w[i];
HT[i].lchild=-1;
HT[i].rchild=-1;
HT[i].parent=-1;
}
for(i=n;i<2*n-1;++i)
{
// HT[i].value=’-1’;
HT[i].weight=-1;
HT[i].lchild=-1;
HT[i].rchild=-1;
HT[i].parent=-1;
}

for(i=n;i<2*n;++i)
{
    min1=Max,min2=Max;
    //找权值最小的两个并记录 它们的下标 
    for(j=0;j<i;++j)
    {
        if(HT[j].weight<min1 && HT[j].parent==-1)
        {
            min2=min1;
            s2=s1;
            min1 =HT[j].weight;
            s1=j;    
        }
        else if(HT[j].weight>min1 && HT[j].weight<min2&& HT[j].parent==-1 )
        {
            min2=HT[j].weight;
            s2=j;     
        }
    }
    HT[i].weight = HT[s1].weight +HT[s2].weight;
    HT[i].lchild = s1;
    HT[i].rchild = s2;
    HT[s1].parent = i;
    HT[s2].parent = i;

}

}
void CreatHuffCode(HuffTree &HT,HuffCode &HC,int n)
{
int i=0,j=0,parent=0,c=0;
char code= NULL;
HC = (HuffCode)malloc((n+1)sizeof(char ));
code = (char)malloc((n+1)*sizeof(char));

for(i=0;i<n;++i)
{
    for(c=0,parent=HT[c].parent,j=0;parent!=-1;c=parent,parent=HT[c].parent)
    {
        if(c==HT[parent].lchild)
        {
            code[j++]='0';
        }
        else if (c==HT[parent].rchild)
        {
            code[j++]='1';
        }
    }
    code[j]='\0';

    HC[i]=(char *)malloc(strlen(code)*sizeof(char));
    strcpy(HC[i],code);  
    free(code);

}

}

{
int i=0;
for(i=0;i<n;++i)
{
cout<<HC[i];
}
