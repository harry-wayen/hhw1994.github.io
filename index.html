2、赋予root权限
 方法一：修改 /etc/sudoers 文件，找到下面一行，把前面的注释（#）去掉
 ## Allows people in group wheel to run all commands
   %wheel    ALL=(ALL)    ALL
  然后修改用户，使其属于root组（wheel），命令如下：
  # usermod -g root tommy
 修改完毕，现在可以用tommy帐号登录，然后用命令 su - ，即可获得root权限进行操作。
 方法二：修改 /etc/sudoers 文件，找到下面一行，在root下面添加一行，如下所示：
 ## Allow root to run any commands anywhere
  root    ALL=(ALL)     ALL
  tommy   ALL=(ALL)     ALL
 修改完毕，现在可以用tommy帐号登录，然后用命令 su - ，即可获得r      oot权限进行操作。

void diamond(int r) {
int x, y;

    for (x=0; x<=2*r; x++) {
        if ( y==x+r || y==x-r || 
                y==-x+r || y==-x+3*r )
            printf("*");
        else
            printf(" ");
    }
    printf("\n");
}

#define Max 32767

typedef struct
{
char value;

}HTNode,HuffTree;
typedef char *HuffCode;

void CreatHuffTree(HuffTree &HT,int w,char a,int n)
{
HT=(HuffTree )malloc(sizeof(HTNode)(2n-1));
int i=0,j=0;
int s1 =0, s2=0;
int min1 = Max,min2 = Max;
for(i=0;i<n;++i)//初始化
{
HT[i].value=a[i];
HT[i].weight=w[i];
HT[i].lchild=-1;
HT[i].rchild=-1;
HT[i].parent=-1;
}
for(i=n;i<2*n-1;++i)
{
// HT[i].value=’-1’;
HT[i].weight=-1;
HT[i].lchild=-1;
HT[i].rchild=-1;
HT[i].parent=-1;
}

for(i=n;i<2*n;++i)
{
    min1=Max,min2=Max;
    //找权值最小的两个并记录 它们的下标 
    for(j=0;j<i;++j)
    {
        if(HT[j].weight<min1 && HT[j].parent==-1)
        {
            min2=min1;
            s2=s1;
            min1 =HT[j].weight;
            s1=j;    
        }
        else if(HT[j].weight>min1 && HT[j].weight<min2&& HT[j].parent==-1 )
        {
            min2=HT[j].weight;
            s2=j;     
        }
    }
    HT[i].weight = HT[s1].weight +HT[s2].weight;
    HT[i].lchild = s1;
    HT[i].rchild = s2;
    HT[s1].parent = i;
    HT[s2].parent = i;

}

}
